\section{試作出下列函數的圖像}
\begin{enumerate}[label={\rm(\arabic *)}]
    \item $\displaystyle y = \rm{sgn}(\sin (x))$
    \item $\displaystyle y = 1 - (x+1)^2$
    \item $\displaystyle y = \sin(x^2)$
\end{enumerate}

  $\displaystyle f(2+x), f(2x), f(x^2), f(f(x)), f(\frac{1}{f(x)})$

\section{用區間表示下列不等式的解}
\begin{enumerate}[label={\rm(\arabic*)}]
    \item $\displaystyle \vert 1 - x \vert  - x \ge 0$
    \item $\displaystyle \left\vert x + \frac{1}{x} \right\vert \le 6 $
    \item $\displaystyle \sin x \ge \frac{\sqrt{2}}{2}$
\end{enumerate}

\section{求下列數集的上下確界，並依據定義加以驗證}
\begin{enumerate}[label={\rm(\arabic*)}]
    \item $\displaystyle S = \left\{x \vert x^2 < 2\right\}$
    \item $\displaystyle S = \left\{x \vert x = n!, n \in \mathbb{N} \right\}$
    \item $\displaystyle S = \left\{x \vert x \in (0,1) \text{and x is irrational number}\right\}$
    \item $\displaystyle S = \left\{x \vert x = 1 - \frac{1}{2^n}, n \in \mathbb{N}\right\}$
\end{enumerate}
